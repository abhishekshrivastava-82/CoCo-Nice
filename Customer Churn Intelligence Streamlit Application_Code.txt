def get_model_versions():
    rows = session.sql(f"SHOW VERSIONS IN MODEL {MODEL}").collect()
    versions = []
    for r in rows:
        versions.append({
            "name": r["name"],
            "aliases": str(r["aliases"]),
            "metadata": r["metadata"],
            "comment": r["comment"],
            "is_default": r["is_default_version"],
        })
    return versions

    version_names = [v["name"] for v in versions]

    for v in versions:
        is_default = "DEFAULT" in v["aliases"]

            meta = json.loads(v["metadata"]) if isinstance(v["metadata"], str) else v["metadata"]
            metrics = meta.get("metrics", {})
            acc = metrics.get("accuracy", "N/A")
            f1 = metrics.get("f1_score", "N/A")
            auc = metrics.get("roc_auc", "N/A")
            if isinstance(acc, float):
                metrics_str = f"Accuracy={acc:.4f}, F1={f1:.4f}, AUC={auc:.4f}"
            else:
                metrics_str = "N/A"
        except:
            metrics_str = "N/A"
        badge = " (DEFAULT)" if is_default else ""
        st.markdown(f"- **{v['name']}{badge}** | {metrics_str}")

    st.markdown("---")

    default_ver = next((v["name"] for v in versions if "DEFAULT" in v["aliases"]), version_names[0])
    selected_version = st.selectbox("Base Version (for reference)", version_names,
                                     index=version_names.index(default_ver),
                                     format_func=lambda x: f"{x} (default)" if x == default_ver else x)

    st.markdown("**Hyperparameters:**")
    hp1, hp2, hp3, hp4 = st.columns(4)
    n_estimators = hp1.number_input("n_estimators", min_value=50, max_value=1000, value=200, step=50)
    max_depth = hp2.number_input("max_depth", min_value=2, max_value=15, value=5, step=1)
    learning_rate = hp3.number_input("learning_rate", min_value=0.01, max_value=1.0, value=0.1, step=0.01, format="%.2f")
    test_size = hp4.number_input("test_size", min_value=0.1, max_value=0.5, value=0.2, step=0.05, format="%.2f")

    next_version_num = len(version_names) + 1

    set_default = st.checkbox("Set as default version after training", value=True)

    if st.button("Retrain & Register New Version", type="primary", use_container_width=True):
        import pandas as pd
        import numpy as np
        from sklearn.model_selection import train_test_split
        from sklearn.metrics import accuracy_score, f1_score, roc_auc_score
        from xgboost import XGBClassifier
        from snowflake.ml.registry import Registry

        with st.spinner("Loading training data..."):
            df = session.table(TRAINING_TABLE).to_pandas()
            X = df[FEATURE_COLS].copy()
            y = df[TARGET_COL].copy()
            X = X.fillna(X.median())

        st.info(f"Training data: {len(X)} samples, {len(FEATURE_COLS)} features, churn rate: {y.mean():.1%}")

        with st.spinner(f"Training XGBoost (n_estimators={n_estimators}, max_depth={max_depth}, lr={learning_rate})..."):
            X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=test_size, random_state=42, stratify=y)

                n_estimators=int(n_estimators),
                max_depth=int(max_depth),
                learning_rate=float(learning_rate),
                subsample=0.8,
                colsample_bytree=0.8,
                random_state=42,
                eval_metric='logloss',
                use_label_encoder=False
            )
            xgb_model.fit(X_train, y_train)

            y_pred = xgb_model.predict(X_test)
            y_proba = xgb_model.predict_proba(X_test)[:, 1]
            acc = accuracy_score(y_test, y_pred)

    def ask_churn_analyst(prompt):
        escaped = prompt.replace("\\", "\\\\").replace("'", "\\'")
        sys_escaped = SYSTEM_PROMPT.replace("\\", "\\\\").replace("'", "\\'")
        sql = f"""
        SELECT SNOWFLAKE.CORTEX.COMPLETE(
            'mistral-large2',
            ARRAY_CONSTRUCT(
                OBJECT_CONSTRUCT('role', 'system', 'content', '{sys_escaped}'),
                OBJECT_CONSTRUCT('role', 'user', 'content', '{escaped}')
            ),
            OBJECT_CONSTRUCT('temperature', 0.1, 'max_tokens', 2000)
        ) AS result
        """
        raw = session.sql(sql).collect()[0]["RESULT"]
        parsed = json.loads(raw)
        content = parsed.get("choices", [{}])[0].get("messages", "")
        if not content:
            content = parsed.get("choices", [{}])[0].get("message", {}).get("content", str(parsed))
        content = content.strip()
        if content.startswith("```"):
            content = re.sub(r"^```(?:json)?\s*", "", content)
            content = re.sub(r"\s*```$", "", content)
        try:
            data = json.loads(content)
        except:
            data = {"explanation": content, "sql": ""}
        return data

    def process_message(prompt):
        st.session_state.messages.append({"role": "user", "content": prompt})
        with st.chat_message("user"):
            st.markdown(prompt)
        with st.chat_message("assistant"):
            with st.spinner("Analyzing..."):
                try:
                    result = ask_churn_analyst(prompt)
                    explanation = result.get("explanation", "")
                    sql_query = result.get("sql", "")
                    assistant_msg = {"role": "assistant"}
                    if explanation:
                        st.markdown(explanation)
                        assistant_msg["content"] = explanation
                    if sql_query:
                        with st.expander("View SQL"):
                            st.code(sql_query, language="sql")
                        assistant_msg["sql"] = sql_query
                        df = session.sql(sql_query).to_pandas()
                        st.dataframe(df, use_container_width=True)
                        assistant_msg["results"] = df
                    st.session_state.messages.append(assistant_msg)
                except Exception as e:
                    err = f"Error: {str(e)}"
                    st.error(err)
                    st.session_state.messages.append({"role": "assistant", "content": err})
