CREATE OR REPLACE TABLE NICE_CALLCENTER_DB.AI_READY.CUSTOMER_INTERACTIONS_DENORM AS
WITH agg AS (
    SELECT
        f.CUSTOMER_ID,
        COUNT(*) AS INTERACTION_COUNT,
        AVG(f.DURATION_SECONDS) AS AVG_DURATION_SEC,
        AVG(f.WRAP_UP_SECONDS) AS AVG_WRAP_UP_SEC,
        AVG(f.TOTAL_HANDLE_TIME_SEC) AS AVG_HANDLE_TIME_SEC,
        SUM(CASE WHEN f.WAS_TRANSFERRED THEN 1 ELSE 0 END) AS TRANSFER_COUNT,
        SUM(CASE WHEN f.WAS_ESCALATED THEN 1 ELSE 0 END) AS ESCALATION_COUNT,
        AVG(f.SENTIMENT_SCORE) AS AVG_SENTIMENT,
        AVG(f.CSAT_SCORE) AS AVG_CSAT,
        AVG(f.NPS_SCORE) AS AVG_NPS,
        AVG(f.CES_SCORE) AS AVG_CES,
        AVG(CASE WHEN f.FIRST_CONTACT_RESOLUTION THEN 1.0 ELSE 0.0 END) AS FCR_RATE
    FROM NICE_CALLCENTER_DB.AI_READY.FACT_INTERACTIONS f
    GROUP BY f.CUSTOMER_ID
)
